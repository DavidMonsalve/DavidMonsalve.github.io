English:


4. Delete from a list of words the words that are in a second ready. Print the original list, the list of words to remove, and the resulting list.


                    
                    x = ["car", "house", "dog", "bread"]
                    y = ["eggs", "car", "house", "ketchup"]
                    z = []
                    
                    print("x (original): ", x)
                    print("y           : ", y)
                    
                    for i in x:
                        if i in y:
                            z.append(i)
                    
                    for i in z:
                        if i in x:
                            x.remove(i)
                    print(" ")
                    print(" ")
                    print("z (a eliminar): ", z)
                    print("x (resultante): ", x)
